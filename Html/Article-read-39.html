jQuery制作遮罩层的代码片段
Tags:JS Reply:80 Time:2012-11-29 Author:admin

<script>
   $('<div id="overlay"></div>')
   .css({
       position    : 'fixed',
       top         : 0,
       left        : 0,
       right       : 0,
       bottom      : 0,
       opacity     : 0.6,
       background  : 'black',
       display     : 'none'
   })
   .appendTo('body')
   .fadeIn('normal')
   .click(function () {
       $(this).fadeOut('normal', function () {
           $(this).remove();
       })
   });
</script>
